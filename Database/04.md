====================================================================================================
MYSQL BETWEEN CLAUSE
====================================================================================================

Definition :
BETWEEN → Filters values within a range (inclusive)

----------------------------------------------------------------------------------------------------
SYNTAX
----------------------------------------------------------------------------------------------------
SELECT column FROM table
WHERE column BETWEEN value1 AND value2;

----------------------------------------------------------------------------------------------------
EQUIVALENT
----------------------------------------------------------------------------------------------------
WHERE age BETWEEN 18 AND 22
= WHERE age >= 18 AND age <= 22

----------------------------------------------------------------------------------------------------
KEY POINT
----------------------------------------------------------------------------------------------------
• Inclusive → includes both limits (18 & 22)

----------------------------------------------------------------------------------------------------
EXAMPLES
----------------------------------------------------------------------------------------------------

-- Numeric
SELECT * FROM student WHERE age BETWEEN 18 AND 22;

-- Date
SELECT * FROM orders
WHERE order_date BETWEEN '2024-02-01' AND '2024-03-01';

-- String (dictionary order)
SELECT * FROM student
WHERE name BETWEEN 'A' AND 'M';

-- NOT BETWEEN
SELECT * FROM student
WHERE age NOT BETWEEN 18 AND 22;
= age < 18 OR age > 22

----------------------------------------------------------------------------------------------------
IMPORTANT CASES
----------------------------------------------------------------------------------------------------
• Order matters :
  BETWEEN 22 AND 18 → EMPTY

• NULL values :
  ignored in results

----------------------------------------------------------------------------------------------------
NOTES
----------------------------------------------------------------------------------------------------
• BETWEEN = >= AND <=
• Works with numbers, dates, strings
• Use for clean range queries

====================================================================================================
MYSQL LIKE CLAUSE
====================================================================================================

Definition :
LIKE → Pattern matching (strings)

----------------------------------------------------------------------------------------------------
SYNTAX
----------------------------------------------------------------------------------------------------
SELECT column FROM table
WHERE column LIKE pattern;

----------------------------------------------------------------------------------------------------
WILDCARDS
----------------------------------------------------------------------------------------------------
% → any characters (0+)
_ → exactly one character

----------------------------------------------------------------------------------------------------
EXAMPLES
----------------------------------------------------------------------------------------------------

-- Starts with
SELECT * FROM student WHERE name LIKE 'A%';

-- Ends with
SELECT * FROM student WHERE name LIKE '%a';

-- Contains
SELECT * FROM student WHERE name LIKE '%aj%';

-- Middle pattern
SELECT * FROM student WHERE name LIKE 'r%l';

-- Surname / suffix
SELECT * FROM student WHERE name LIKE '% sharma';

-- Single character
SELECT * FROM student WHERE name LIKE '_a%';

-- Exact length (4 chars)
SELECT * FROM student WHERE name LIKE '____';

-- Multiple conditions
SELECT * FROM student
WHERE name LIKE 'A%' AND city LIKE '%i';

-- NOT LIKE
SELECT * FROM student WHERE name NOT LIKE 'A%';

----------------------------------------------------------------------------------------------------
CASE SENSITIVITY
----------------------------------------------------------------------------------------------------
• Depends on collation (usually case-insensitive)

----------------------------------------------------------------------------------------------------
NOTES
----------------------------------------------------------------------------------------------------
• Used only for strings
• Slower than '=' (may skip index)
• Used in search/filter features

====================================================================================================
MYSQL IN CLAUSE
====================================================================================================

Definition :
IN → Matches any value in a list (shortcut for OR)

----------------------------------------------------------------------------------------------------
SYNTAX
----------------------------------------------------------------------------------------------------
SELECT column FROM table
WHERE column IN (v1, v2, v3);

----------------------------------------------------------------------------------------------------
EQUIVALENT
----------------------------------------------------------------------------------------------------
city IN ('Delhi','Mumbai')
= city='Delhi' OR city='Mumbai'

----------------------------------------------------------------------------------------------------
EXAMPLES
----------------------------------------------------------------------------------------------------
-- String
SELECT * FROM student WHERE name IN ('Ajay','Rahul');

-- Numeric
SELECT * FROM student WHERE age IN (18,20,22);

-- NOT IN
SELECT * FROM student WHERE city NOT IN ('Delhi','Mumbai');

-- Subquery
SELECT * FROM student
WHERE id IN (SELECT student_id FROM orders);

----------------------------------------------------------------------------------------------------
IMPORTANT CASE (NULL)
----------------------------------------------------------------------------------------------------
SELECT * FROM student WHERE age NOT IN (18,20,NULL);
→ EMPTY (unexpected)

Reason :
NULL → comparison becomes UNKNOWN

✔ Avoid NULL in IN list

----------------------------------------------------------------------------------------------------
NOTES
----------------------------------------------------------------------------------------------------
• IN = multiple OR
• Improves readability
• Use carefully with NULL

====================================================================================================
MYSQL IS NULL
====================================================================================================

Definition :
IS NULL → Checks missing values

----------------------------------------------------------------------------------------------------
SYNTAX
----------------------------------------------------------------------------------------------------
SELECT * FROM table WHERE column IS NULL;
SELECT * FROM table WHERE column IS NOT NULL;

----------------------------------------------------------------------------------------------------
EXAMPLE
----------------------------------------------------------------------------------------------------
SELECT * FROM student WHERE age IS NULL;

----------------------------------------------------------------------------------------------------
COMMON MISTAKE
----------------------------------------------------------------------------------------------------
WRONG  : age = NULL / age != NULL
RIGHT  : age IS NULL / IS NOT NULL

----------------------------------------------------------------------------------------------------
WHY
----------------------------------------------------------------------------------------------------
• NULL = unknown → cannot compare using =

----------------------------------------------------------------------------------------------------
COMBINED CONDITIONS
----------------------------------------------------------------------------------------------------
SELECT * FROM student
WHERE age IS NULL OR city = 'Delhi';

----------------------------------------------------------------------------------------------------
NOTES
----------------------------------------------------------------------------------------------------
• NULL ≠ 0 / empty string
• Ignored in comparisons
• Very important in real data

====================================================================================================
MYSQL ALIAS (AS)
====================================================================================================

Definition :
Alias → Temporary name for column/table

----------------------------------------------------------------------------------------------------
SYNTAX
----------------------------------------------------------------------------------------------------
SELECT column AS alias FROM table;

----------------------------------------------------------------------------------------------------
COLUMN ALIAS
----------------------------------------------------------------------------------------------------
SELECT name AS student_name FROM student;

-- Without AS (valid)
SELECT name student_name FROM student;

-- Multiple
SELECT name AS n, age AS a FROM student;

-- Expression
SELECT name, age + 10 AS future_age FROM student;

----------------------------------------------------------------------------------------------------
TABLE ALIAS
----------------------------------------------------------------------------------------------------
SELECT s.name, s.age
FROM student AS s;

----------------------------------------------------------------------------------------------------
JOIN EXAMPLE
----------------------------------------------------------------------------------------------------
SELECT s.name, o.order_date
FROM student AS s
JOIN orders AS o
ON s.id = o.student_id;

----------------------------------------------------------------------------------------------------
RULES
----------------------------------------------------------------------------------------------------
• Exists only during query
• Cannot use in WHERE (same level)
• Can use in ORDER BY

----------------------------------------------------------------------------------------------------
EXAMPLE
----------------------------------------------------------------------------------------------------
SELECT name AS n
FROM student
ORDER BY n;

----------------------------------------------------------------------------------------------------
NOTES
----------------------------------------------------------------------------------------------------
• Improves readability
• Useful in joins & calculations
• Avoid spaces (or use quotes)
• Temporary (not stored)

====================================================================================================
MYSQL STRING FUNCTIONS
====================================================================================================

Definition :
String functions → Used to manipulate/process text data

Concept :
char  → character (A, B)
byte  → storage unit (memory)

====================================================================================================
CORE FUNCTIONS
====================================================================================================

1) LENGTH()
----------------------------------------------------------------------------------------------------
Returns length in BYTES

SELECT LENGTH('Ajay');         → 4
SELECT LENGTH('🙂');           → depends on bytes (multi-byte)

Use case : validation (password, phone length)

----------------------------------------------------------------------------------------------------
2) CHAR_LENGTH()
----------------------------------------------------------------------------------------------------
Returns number of characters

SELECT CHAR_LENGTH('Ajay');    → 4

----------------------------------------------------------------------------------------------------
3) UPPER() / LOWER()
----------------------------------------------------------------------------------------------------
SELECT UPPER('ajay');          → 'AJAY'
SELECT LOWER('AJAY');          → 'ajay'

----------------------------------------------------------------------------------------------------
4) CONCAT()
----------------------------------------------------------------------------------------------------
Joins strings

SELECT CONCAT('Ajay',' ','Sharma');     → 'Ajay Sharma'
SELECT CONCAT('Age: ', 25);             → 'Age: 25'

⚠ NULL case :
CONCAT('Ajay', NULL, 'Sharma') → NULL

----------------------------------------------------------------------------------------------------
5) CONCAT_WS()  (WITH SEPARATOR)
----------------------------------------------------------------------------------------------------
SELECT CONCAT_WS('-', '2024','03','20');   → '2024-03-20'

✔ Ignores NULL values :
CONCAT_WS('_','Ajay',NULL,'Sharma') → 'Ajay_Sharma'

----------------------------------------------------------------------------------------------------
6) SUBSTRING() / SUBSTR()
----------------------------------------------------------------------------------------------------
Extract part of string

SELECT SUBSTRING('AjaySharma',1,4);   → 'Ajay'
SELECT SUBSTRING('AjaySharma',5);     → 'Sharma'

• Index starts from 1 (not 0)

----------------------------------------------------------------------------------------------------
7) LEFT() / RIGHT()
----------------------------------------------------------------------------------------------------
SELECT LEFT('AjaySharma',4);   → 'Ajay'
SELECT RIGHT('AjaySharma',6);  → 'Sharma'

Use cases :
• File extension → RIGHT('file.pdf',3)
• Masking → CONCAT('******', RIGHT('1234567890',4))

----------------------------------------------------------------------------------------------------
8) TRIM()
----------------------------------------------------------------------------------------------------
SELECT TRIM('  Ajay  ');       → 'Ajay'

----------------------------------------------------------------------------------------------------
9) LTRIM() / RTRIM()
----------------------------------------------------------------------------------------------------
SELECT LTRIM('  Ajay');        → 'Ajay'
SELECT RTRIM('Ajay  ');        → 'Ajay'

----------------------------------------------------------------------------------------------------
10) REPLACE()
----------------------------------------------------------------------------------------------------
Replace substring

SELECT REPLACE('Hello World','World','Ajay');   → 'Hello Ajay'
SELECT REPLACE('apple apple','apple','mango');  → 'mango mango'

• Case-sensitive
• If not found → original string returned

====================================================================================================
SEARCH FUNCTIONS
====================================================================================================

11) INSTR()
----------------------------------------------------------------------------------------------------
Returns position of FIRST occurrence

SYNTAX :
INSTR(string, substring)

SELECT INSTR('Hello Ajay','Ajay');   → 7

----------------------------------------------------------------------------------------------------
KEY POINTS
----------------------------------------------------------------------------------------------------
• Index starts from 1
• Not found → 0
• First occurrence only
• Empty string → returns 1
• Case depends on collation

----------------------------------------------------------------------------------------------------
USAGE
----------------------------------------------------------------------------------------------------
SELECT * FROM student
WHERE INSTR(name,'aj') > 0;

----------------------------------------------------------------------------------------------------
12) LOCATE()
----------------------------------------------------------------------------------------------------
Similar to INSTR but more flexible

SYNTAX :
LOCATE(substring, string)
LOCATE(substring, string, start_pos)

SELECT LOCATE('Ajay','Hello Ajay');     → 7
SELECT LOCATE('Ajay','Ajay Ajay',2);    → 6

----------------------------------------------------------------------------------------------------
KEY DIFFERENCE
----------------------------------------------------------------------------------------------------
INSTR(string, substring)
LOCATE(substring, string [, start])

✔ LOCATE supports start position

----------------------------------------------------------------------------------------------------
COMMON USE CASE (EMAIL)
----------------------------------------------------------------------------------------------------
LOCATE('@', email) → position of '@'

Username extraction :
SELECT SUBSTRING(email,1,LOCATE('@',email)-1) FROM customers;

----------------------------------------------------------------------------------------------------
EDGE CASE
----------------------------------------------------------------------------------------------------
If '@' not present → LOCATE = 0 → result invalid

✔ Safe version :
CASE WHEN LOCATE('@',email)>0
THEN SUBSTRING(email,1,LOCATE('@',email)-1)
END

----------------------------------------------------------------------------------------------------
MEMORY TRICK
----------------------------------------------------------------------------------------------------
LOCATE → position
-1     → length before it

====================================================================================================
OTHER STRING FUNCTIONS
====================================================================================================

13) REVERSE()
----------------------------------------------------------------------------------------------------
SELECT REVERSE('Ajay');   → 'yajA'

----------------------------------------------------------------------------------------------------
14) LPAD() / RPAD()
----------------------------------------------------------------------------------------------------
SELECT LPAD('Ajay',6,'*');   → '**Ajay'
SELECT RPAD('Ajay',6,'*');   → 'Ajay**'

----------------------------------------------------------------------------------------------------
15) FORMAT()
----------------------------------------------------------------------------------------------------
SELECT FORMAT(12345.678,2);   → '12,345.68'

----------------------------------------------------------------------------------------------------
16) ASCII()
----------------------------------------------------------------------------------------------------
SELECT ASCII('A');   → 65

----------------------------------------------------------------------------------------------------
17) CHAR()
----------------------------------------------------------------------------------------------------
SELECT CHAR(65);     → 'A'

----------------------------------------------------------------------------------------------------
18) FIND_IN_SET()
----------------------------------------------------------------------------------------------------
SELECT FIND_IN_SET('Ajay','Rahul,Ajay,Amit');   → 2

====================================================================================================
IMPORTANT NOTES
====================================================================================================
• Strings → 'single quotes'
• Index starts from 1
• LENGTH vs CHAR_LENGTH → important for Unicode
• CONCAT vs CONCAT_WS → NULL handling
• Functions can be combined
• Used in SELECT / WHERE / ORDER BY
