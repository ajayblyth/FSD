----------------------------------------------------------------------
UPPER() / LOWER()
----------------------------------------------------------------------
DEFINITION:
Convert string case

SYNTAX:
UPPER(str)       | LOWER(str)

EXAMPLES:
SELECT UPPER('ajay');           → 'AJAY'
SELECT LOWER('AJAY');           → 'ajay'

TABLE USAGE:
SELECT UPPER(name) FROM student;
SELECT LOWER(email) FROM users;

WHERE (CASE-INSENSITIVE SEARCH):
SELECT * FROM student WHERE UPPER(name) = 'AJAY';

REAL USE:
SELECT * FROM users WHERE LOWER(email) = 'ajay@gmail.com';

INSERT NORMALIZATION:
INSERT INTO users(email) VALUES (LOWER('AJAY@GMAIL.COM'));

NOTES:
• Output only (does NOT change stored data)
• Helps in consistent comparison
• May skip index → slower on large data

======================================================================

----------------------------------------------------------------------
TRIM() / LTRIM() / RTRIM()
----------------------------------------------------------------------
DEFINITION:
Removes unwanted characters (default: spaces)

SYNTAX:
TRIM(str)
TRIM(char FROM str)

EXAMPLES:
SELECT TRIM('  Ajay  ');              → 'Ajay'
SELECT TRIM('x' FROM 'xxxAjayxxx');  → 'Ajay'

LEFT / RIGHT TRIM:
TRIM(LEADING 'x' FROM 'xxxAjay')     → 'Ajay'
TRIM(TRAILING 'x' FROM 'Ajayxxx')    → 'Ajay'

TABLE:
SELECT TRIM(name) FROM student;

NOTES:
• Removes only start/end (not middle)
• Useful for cleaning user input
• Output only

======================================================================

----------------------------------------------------------------------
REVERSE()
----------------------------------------------------------------------
DEFINITION:
Reverses string

SYNTAX:
REVERSE(str)

EXAMPLES:
SELECT REVERSE('Ajay');          → 'yajA'
SELECT REVERSE(name) FROM student;

USE CASE:
• Reverse string
• Extract last parts

COMBINATION:
SELECT REVERSE(SUBSTRING('AjaySharma',1,4)); → 'yajA'

NOTES:
• Output only
• Works on strings

----------------------------------------------------------------------
FORMAT()
----------------------------------------------------------------------
DEFINITION:
Formats number with commas + decimals (returns STRING)

SYNTAX:
FORMAT(number, decimals)

EXAMPLES:
SELECT FORMAT(12345.678, 2);   → '12,345.68'
SELECT FORMAT(1000000, 0);     → '1,000,000'

NOTES:
• Output is STRING (not numeric)
• Used only for display (not calculation)

Note:
• Commas are placed using standard thousands grouping
• Pattern → every 3 digits from right side

---------------------------------------------------------------
EXAMPLES
---------------------------------------------------------------

SELECT FORMAT(1234, 0);        → '1,234'
SELECT FORMAT(12345, 0);       → '12,345'
SELECT FORMAT(1234567, 0);     → '1,234,567'

MYSQL ARITHMETIC OPERATORS
======================================================================

----------------------------------------------------------------------
OPERATORS
----------------------------------------------------------------------
+   Add      | -   Subtract
*   Multiply | /   Divide
%   Modulus  | DIV Integer division

----------------------------------------------------------------------
BASIC
----------------------------------------------------------------------
SELECT 2 + 3;      → 5
SELECT 10 / 2;     → 5  Quotient (result of division)
SELECT 10 % 3;     → 1   Remainder
SELECT 10 DIV 3;   → 3

----------------------------------------------------------------------
TABLE USAGE
----------------------------------------------------------------------
SELECT name, age + 5 FROM student;
SELECT price * quantity FROM orders;

ALIAS:
SELECT price * quantity AS total_amount FROM orders;

----------------------------------------------------------------------
WHERE
----------------------------------------------------------------------
SELECT * FROM student WHERE age + 5 > 25;

----------------------------------------------------------------------
NULL BEHAVIOR
----------------------------------------------------------------------
SELECT 5 + NULL;   → NULL
SELECT 10 * NULL;  → NULL

→ Any operation with NULL → NULL

----------------------------------------------------------------------
PRECEDENCE
----------------------------------------------------------------------
1) * / %
2) + -

SELECT 2 + 3 * 4;   → 14
SELECT (2+3) * 4;   → 20

======================================================================
EXAMPLE TABLE
======================================================================

CREATE TABLE cart(
  id INT PRIMARY KEY AUTO_INCREMENT,
  item_name VARCHAR(30),
  item_price INT,
  item_qty INT
);

SELECT item_name, item_price, item_qty,
       item_price * item_qty AS total_price
FROM cart;

======================================================================
DISCOUNT CALCULATION
======================================================================

FORMULA:
Final Price = price - (price * discount / 100)

EXAMPLE:
SELECT 1000 - (1000 * 10 / 100); → 900

TABLE:
SELECT name, price, discount,
       price - (price * discount / 100) AS final_price
FROM products;

DISCOUNT ONLY:
SELECT price * discount / 100 AS discount_amount FROM products;

FILTER:
SELECT * FROM products
WHERE price - (price * discount / 100) < 500;

======================================================================
ARITHMETIC EDGE CASES
======================================================================

1) DIVISION BY ZERO
SELECT 10 / 0; → ERROR / NULL

✔ Avoid:
WHERE denominator != 0

----------------------------------------------------------------------
2) NULL VALUES
SELECT 10 + NULL → NULL

----------------------------------------------------------------------
3) STRING + NUMBER
SELECT '10' + 5   → 15
SELECT 'abc' + 5  → 5   ('abc' → 0)

• MySQL reads number from start of string
• '20abc' → 20 (stops at first non-numeric character)
• So: 10 + 20 = 30


SELECT 10 + 'abc20';   → 10
• 'abc20' → starts with non-numeric → converted to 0
• So: 10 + 0 = 10

• MySQL automatically converts string(number should be inside) to number in arithmetic
• '10' → 10, so 10 + 5 = 15
• Non-numeric string ('abc') → converted to 0 → 0 + 5 = 5

----------------------------------------------------------------------
4) NEGATIVE
SELECT -10 + 5 → -5

----------------------------------------------------------------------
5) FLOAT PRECISION
SELECT 10 / 3 → 3.3333 (approx)

----------------------------------------------------------------------
6) INTEGER DIVISION
SELECT 10 DIV 3 → 3

----------------------------------------------------------------------
7) MOD BY ZERO
SELECT 10 % 0 → ERROR

----------------------------------------------------------------------
8) OVERFLOW
Very large numbers → incorrect / overflow

----------------------------------------------------------------------
9) ORDER
Always use () for clarity

----------------------------------------------------------------------
10) MIXED TYPES
SELECT 10 + '20abc' → 30

----------------------------------------------------------------------
NOTES
----------------------------------------------------------------------
• Validate data before calculation
• Handle NULL & zero carefully
• Use () for clarity
• Watch type conversion

MYSQL ALTER TABLE (MODIFYING TABLE STRUCTURE)
======================================================================

DEFINITION:
ALTER TABLE → used to modify existing table structure (columns, constraints, etc.)

======================================================================
1) ADD COLUMN
======================================================================
ALTER TABLE student 
ADD age INT;

ALTER TABLE student 
ADD salary INT NOT NULL;
→ NOT NULL → default becomes 0 (if no value provided), else NULL if allowed

----------------------------------------------------------------------
ADD MULTIPLE COLUMNS
----------------------------------------------------------------------
ALTER TABLE student 
ADD (city VARCHAR(50), marks INT);

======================================================================
2) MODIFY COLUMN (DATATYPE / SIZE / CONSTRAINT)
======================================================================
DEFINITION:
MODIFY → change datatype, size, or constraints of existing column

SYNTAX:
ALTER TABLE table_name 
MODIFY column_name new_definition;

----------------------------------------------------------------------
EXAMPLES
----------------------------------------------------------------------
ALTER TABLE student MODIFY name VARCHAR(100);         → increase size
ALTER TABLE student MODIFY age BIGINT;                → INT → BIGINT
ALTER TABLE student MODIFY name VARCHAR(50) NOT NULL; → add NOT NULL
ALTER TABLE student MODIFY name VARCHAR(50) NULL;     → remove NOT NULL

----------------------------------------------------------------------
WITH POSITION
----------------------------------------------------------------------
ALTER TABLE student MODIFY name VARCHAR(100) AFTER id;

----------------------------------------------------------------------
IMPORTANT CASES
----------------------------------------------------------------------
• Data loss risk       → VARCHAR(100) → VARCHAR(10) (data truncated ❌)
• Incompatible type    → VARCHAR → INT fails if non-numeric data ❌
• Default reset        → must re-specify DEFAULT if needed
• NULL values          → remain NULL (no conversion)

----------------------------------------------------------------------
NOTES
----------------------------------------------------------------------
1) Must re-define full column (type + constraints)
2) Affects structure only (data remains)
3) Backup recommended

======================================================================
3) CHANGE COLUMN (RENAME + MODIFY)
======================================================================
DEFINITION:
CHANGE → rename column (and optionally modify datatype)

SYNTAX:
ALTER TABLE table_name
CHANGE old_name new_name datatype;

----------------------------------------------------------------------
EXAMPLES
----------------------------------------------------------------------
ALTER TABLE student CHANGE name full_name VARCHAR(50);      → rename
ALTER TABLE student CHANGE age student_age BIGINT;          → rename + type
ALTER TABLE student CHANGE name full_name VARCHAR(50) AFTER id;

----------------------------------------------------------------------
IMPORTANT RULE
----------------------------------------------------------------------
• Datatype MUST be specified (even if unchanged)

----------------------------------------------------------------------
MODERN (MYSQL 8+) RENAME COLUMN  TO
----------------------------------------------------------------------
ALTER TABLE student RENAME COLUMN name TO full_name;

----------------------------------------------------------------------
COMMON MISTAKE
----------------------------------------------------------------------
ALTER TABLE student CHANGE name full_name;   ❌ (datatype missing)

----------------------------------------------------------------------
NOTES
----------------------------------------------------------------------
1) CHANGE = rename + redefine
2) Use RENAME COLUMN if only renaming

======================================================================
4) DROP COLUMN
======================================================================
ALTER TABLE student DROP COLUMN age;

-- Multiple columns
ALTER TABLE student DROP COLUMN age, DROP COLUMN salary;

----------------------------------------------------------------------
NOTES
----------------------------------------------------------------------
• Data permanently deleted ❌ (backup needed)
• Cannot drop PK / FK without handling dependencies

======================================================================
5) PRIMARY KEY OPERATIONS
======================================================================
DROP PRIMARY KEY:
ALTER TABLE student DROP PRIMARY KEY;

----------------------------------------------------------------------
RULES / EDGE CASES
----------------------------------------------------------------------
• Only one PK per table
• AUTO_INCREMENT present → must remove first:

  ALTER TABLE student MODIFY id INT;  -- removes AUTO_INCREMENT
  ALTER TABLE student DROP PRIMARY KEY;

• FK dependency exists → drop FK first ❌
• Data remains, only constraint removed
• Table will have no unique identifier (not recommended)

IF FOREIGN KEY DEPENDENCY EXISTS ❌
---------------------------------------------------------------

• First drop foreign key from child table

-- Find FK name
SHOW CREATE TABLE child_table;

-- Drop FK
ALTER TABLE child_table 
DROP FOREIGN KEY fk_name;
======================================================================
6) ADD CONSTRAINT
======================================================================
DEFINITION:
Add rules (constraints) to existing columns


ALTER TABLE student 
ADD CONSTRAINT unique_email UNIQUE (email);

-- Shortcut
ALTER TABLE student ADD UNIQUE(email);

----------------------------------------------------------------------
NOT NULL (via MODIFY)
----------------------------------------------------------------------
ALTER TABLE people MODIFY name VARCHAR(30) NOT NULL;
ALTER TABLE people MODIFY name VARCHAR(30) NULL;

----------------------------------------------------------------------
OTHER EXAMPLES
----------------------------------------------------------------------
ALTER TABLE student ADD PRIMARY KEY (id);

ALTER TABLE orders
ADD CONSTRAINT fk_student
FOREIGN KEY (student_id) REFERENCES student(id);

ALTER TABLE student MODIFY city VARCHAR(50) DEFAULT 'Delhi';

----------------------------------------------------------------------
IMPORTANT DIFFERENCE
----------------------------------------------------------------------
UNIQUE     → ADD CONSTRAINT / ADD UNIQUE
NOT NULL   → MODIFY COLUMN

----------------------------------------------------------------------
EDGE CASES
----------------------------------------------------------------------
• UNIQUE fails if duplicates exist ❌
• NOT NULL fails if NULL values exist ❌
• FK fails if parent data missing ❌

----------------------------------------------------------------------
CHECK BEFORE UNIQUE
----------------------------------------------------------------------
SELECT name, COUNT(*)
FROM people
GROUP BY name
HAVING COUNT(*) > 1;

----------------------------------------------------------------------
NOTES
----------------------------------------------------------------------
1) UNIQUE creates index internally
2) Multiple UNIQUE allowed
3) Naming constraint helps in DROP

======================================================================
7) DROP CONSTRAINT
======================================================================
ALTER TABLE student DROP INDEX unique_email;

======================================================================
8) RENAME TABLE ---- TO ----
======================================================================
RENAME TABLE student TO students;

======================================================================
9) DEFAULT VALUE
======================================================================
SET DEFAULT:
ALTER TABLE student 
ALTER city SET DEFAULT 'Delhi';

REMOVE DEFAULT:
ALTER TABLE student 
ALTER city DROP DEFAULT;

======================================================================
10) COLUMN POSITION (AFTER / FIRST)
======================================================================
DEFINITION:
Controls column order while adding/modifying

----------------------------------------------------------------------
EXAMPLES
----------------------------------------------------------------------
ALTER TABLE student ADD city VARCHAR(50) AFTER name; → specific position
ALTER TABLE student ADD id INT FIRST;                → beginning
ALTER TABLE student ADD age INT;                     → default (end)

----------------------------------------------------------------------
NOTES
----------------------------------------------------------------------
1) Only affects column order (not data)
2) AFTER / FIRST optional
3) Mainly used with ADD COLUMN

======================================================================
FINAL NOTES
======================================================================
1) ALTER TABLE → structure changes only (not data directly)
2) Changes are permanent ⚠
3) Large tables → slower operations
4) Always take backup before major changes

======================================================================
MYSQL CRUD OPERATIONS
======================================================================

DEFINITION:
CRUD → Create, Read, Update, Delete (basic data operations)

======================================================================
1) CREATE (INSERT)
======================================================================
INSERT INTO student (id, name, age)
VALUES (1, 'Ajay', 22);

→ Adds new row

======================================================================
2) READ (SELECT)
======================================================================
SELECT * FROM student;
SELECT name, age FROM student;

→ Retrieves data

======================================================================
3) UPDATE
======================================================================
UPDATE student
SET age = 23
WHERE id = 1;

→ Updates specific row

----------------------------------------------------------------------
MULTIPLE COLUMNS
----------------------------------------------------------------------
UPDATE student
SET age = 23, username = 'robin_usa'
WHERE id = 1;

----------------------------------------------------------------------
WITHOUT WHERE ⚠
----------------------------------------------------------------------
UPDATE student
SET age = age + 1;

→ Updates ALL rows ❌

Note: MySQL safe mode may warn/block this

----------------------------------------------------------------------
NULL CASE
----------------------------------------------------------------------
UPDATE student
SET country = 'IND'
WHERE country IS NULL;

======================================================================
4) DELETE
======================================================================
DELETE FROM student
WHERE id = 1;

→ Deletes specific row

======================================================================
DELETE + AUTO_INCREMENT BEHAVIOR
======================================================================

EXAMPLE:
id | name
---------
1  | Ajay
2  | Rahul
3  | Amit

DELETE FROM student
WHERE id = 2;

1  | Ajay
3  | Amit

----------------------------------------------------------------------
INSERT AFTER DELETE
----------------------------------------------------------------------
INSERT INTO student (name) VALUES ('Neha');

→ id = 4 ✔ (NOT 2)

----------------------------------------------------------------------
KEY POINT
----------------------------------------------------------------------
AUTO_INCREMENT does NOT reuse deleted ids

----------------------------------------------------------------------
RESET AUTO_INCREMENT (OPTIONAL)
----------------------------------------------------------------------
ALTER TABLE student AUTO_INCREMENT = 2;

→ Works only if value is free

----------------------------------------------------------------------
NOTES
----------------------------------------------------------------------
1) Gaps in id are normal
2) Do not expect continuous numbering
3) Ensures uniqueness & consistency

======================================================================
IMPORTANT NOTES (CRUD)
======================================================================
1) WHERE is critical in UPDATE / DELETE ⚠
2) SELECT → no data change
3) INSERT → adds new rows
4) UPDATE → modifies existing rows
5) DELETE → removes data permanently

MYSQL ORDER BY
======================================================================

DEFINITION:
ORDER BY → sorts query result

----------------------------------------------------------------------
BASIC
----------------------------------------------------------------------
SELECT column_name
FROM table_name
ORDER BY column_name;

----------------------------------------------------------------------
ASC (DEFAULT)
----------------------------------------------------------------------
SELECT * FROM student
ORDER BY age;

→ small → large

----------------------------------------------------------------------
DESC
----------------------------------------------------------------------
SELECT * FROM student
ORDER BY age DESC;

→ large → small

----------------------------------------------------------------------
MULTIPLE COLUMNS
----------------------------------------------------------------------
SELECT * FROM student
ORDER BY city ASC, age DESC;

----------------------------------------------------------------------
COLUMN POSITION
----------------------------------------------------------------------
SELECT name, age FROM student
ORDER BY 2;

→ sort by 2nd column (age)

----------------------------------------------------------------------
WITH WHERE
----------------------------------------------------------------------
SELECT * FROM student
WHERE age > 20
ORDER BY age;

----------------------------------------------------------------------
WITH ALIAS
----------------------------------------------------------------------
SELECT name, age + 5 AS future_age
FROM student
ORDER BY future_age;

----------------------------------------------------------------------
NULL BEHAVIOR
----------------------------------------------------------------------
ASC (small → large / A → Z)  → NULL first
DESC(large → small / Z → A) → NULL last

----------------------------------------------------------------------
NOTES
----------------------------------------------------------------------
1) Default → ASC
2) Used after WHERE
3) Improves readability
======================================================================
MYSQL LIMIT
======================================================================

DEFINITION:
LIMIT → restrict number of rows returned

----------------------------------------------------------------------
BASIC
----------------------------------------------------------------------
SELECT * FROM student
LIMIT 5;

→ first 5 rows

----------------------------------------------------------------------
OFFSET
----------------------------------------------------------------------
SELECT * FROM student
LIMIT 3 OFFSET 2;

→ same as above

----------------------------------------------------------------------
WITH ORDER BY
----------------------------------------------------------------------
SELECT * FROM student
ORDER BY age DESC
LIMIT 3;

→ top 3 values

----------------------------------------------------------------------
USE CASES
----------------------------------------------------------------------
• Pagination
• Top N records
• Testing queries

----------------------------------------------------------------------
NOTES
----------------------------------------------------------------------
1) LIMIT at end
2) Default offset = 0
3) Use with ORDER BY for accuracy
4) Improves performance
